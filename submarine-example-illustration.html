Here is what Bayes search looks like!

Prior probability

The black point in the figure below indicates where the submarine actually is.

rm(list=ls(all=TRUE))
setwd('U:\\uf\\courses\\bayesian course\\example submarine')
source('submarine aux.R')
set.seed(1)

#create prior probability for location
nloc=5
seq1=1:nloc
combo=expand.grid(x=seq1,y=seq1)
combo$prob=dnorm(combo$x,mean=3,sd=2)*dnorm(combo$y,mean=3,sd=2)
combo$prob=combo$prob/sum(combo$prob)

#visual depiction of prior probabilities
pi1=create.heatmap(combo,nloc)

par(mfrow=c(1,1),ask=T)
image(pi1,x=1:nloc,y=1:nloc,xlab='',ylab='')
text(x=combo$x,y=combo$y,round(combo$prob,3),pos=1)

#location of submarine
ind=rmultinom(1,size=1,prob=combo$prob)
ind1=which(ind==1)
x.submarine=combo$x[ind1]
y.submarine=combo$y[ind1]
points(x=x.submarine,y=y.submarine,pch=19)

Updating probabilities

The blue circles in the figure below indicate where we are searching/have already searched

#probability of finding given presence
theta=0.1
nsteps=100

#to store intermediate outcomes
pontos=matrix(NA,nsteps,2); colnames(pontos)=c('x','y')

par(ask=T)
for (i in 1:nsteps){
  ind=which(combo$prob==max(combo$prob))
  if (length(ind)>1) ind=sample(ind,size=1)
  xsel=combo$x[ind]; ysel=combo$y[ind]
  
  #if you choose the location where the submarine actually is
  if (xsel==x.submarine & ysel==y.submarine){
    z=rbinom(1,size=1,prob=theta)
    if (z==1){
      print('You found it!')
      break;
    }
  }
  combo=update.probs(combo=combo,xsel=xsel,ysel=ysel,theta=theta)
  pi1=create.heatmap(combo=combo,nloc=nloc)
  image(pi1,x=1:nloc,y=1:nloc,main=paste('Step',i),xlab='',ylab='')
  text(x=combo$x,y=combo$y,round(combo$prob,3),pos=1)
  points(x=x.submarine,y=y.submarine,pch=19)
  
  pontos[i,]=c(xsel,ysel)
  lines(x=pontos[,'x'],y=pontos[,'y'],col='blue')
  points(x=pontos[,'x'],y=pontos[,'y'],col='blue')
}

The code for the functions “update.probs” and “create.heatmap” are available in the file “submarine aux.R” and are shown below:

#file "submarine aux.R"
update.probs=function(combo,xsel,ysel,theta){
  cond=combo$x==xsel & combo$y==ysel
  num1=(1-theta)*combo$prob[cond]
  den1=num1+(1-combo$prob[cond])
  prob.loc=num1/den1
  combo$prob=combo$prob/den1
  combo$prob[cond]=prob.loc
  combo
}
create.heatmap=function(combo,nloc){
  pi1=matrix(NA,nloc,nloc)
  for (i in 1:nrow(combo)){
    pi1[combo$x[i],combo$y[i]]=combo$prob[i]  
  }
  pi1
}
