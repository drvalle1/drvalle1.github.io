Which strategy works best for finding the USS Scorpion submarine?

Strategy 1: random search

rm(list=ls(all=TRUE))
set.seed(1)

#create prior probability for location
nloc=20
seq1=1:nloc
combo=expand.grid(x=seq1,y=seq1)
combo$prob=dnorm(combo$x,mean=3,sd=2)*dnorm(combo$y,mean=3,sd=2)
combo$prob=combo$prob/sum(combo$prob)

#some useful things
nsim=1000
nsteps=10000

#probability of finding given presence
theta=0.2

#to store results
res.random=rep(NA,nsim)
for (i in 1:nsim){
  #location of submarine
  ind=rmultinom(1,size=1,prob=combo$prob)
  ind1=which(ind==1)
  x.submarine=combo$x[ind1]
  y.submarine=combo$y[ind1]

  for (j in 1:nsteps){
    #choose a location at random
    ind=sample(1:nrow(combo),size=1)
    xsel=combo$x[ind]; ysel=combo$y[ind]
    
    #did you find it?
    if (xsel==x.submarine & ysel==y.submarine){
      z=rbinom(1,size=1,prob=theta)
      if (z==1){
        res.random[i]=j
        break;
      }
    }
  }  
}

Strategy 3: search the sector that has highest posterior probability

setwd('U:\\uf\\courses\\bayesian course\\submarine example')
source('submarine aux.R')
set.seed(1)

#create prior probability for location
nloc=20
seq1=1:nloc

#some useful things
nsim=1000
nsteps=10000

#probability of finding given presence
theta=0.2

#to store results
res.bayes=rep(NA,nsim)
for (i in 1:nsim){
  #location of submarine
  combo=expand.grid(x=seq1,y=seq1)
  combo$prob=dnorm(combo$x,mean=3,sd=2)*dnorm(combo$y,mean=3,sd=2)
  combo$prob=combo$prob/sum(combo$prob)
  ind=rmultinom(1,size=1,prob=combo$prob)
  ind1=which(ind==1)
  x.submarine=combo$x[ind1]
  y.submarine=combo$y[ind1]

  for (j in 1:nsteps){
    ind=which(combo$prob==max(combo$prob))
    if (length(ind)>1) ind=sample(ind,size=1)
    xsel=combo$x[ind]; ysel=combo$y[ind]
  
    #if you choose the location where the submarine actually is
    if (xsel==x.submarine & ysel==y.submarine){
      z=rbinom(1,size=1,prob=theta)
      if (z==1){
        res.bayes[i]=j
        break;
      }
    }
    combo=update.probs(combo=combo,xsel=xsel,ysel=ysel,theta=theta)
  }  
}

This code requires the auxiliary functions given below to work.

#file 'submarine aux.R'
update.probs=function(combo,xsel,ysel,theta){
  cond=combo$x==xsel & combo$y==ysel
  num1=(1-theta)*combo$prob[cond]
  den1=num1+(1-combo$prob[cond])
  prob.loc=num1/den1
  combo$prob=combo$prob/den1
  combo$prob[cond]=prob.loc
  combo
}
create.heatmap=function(combo,nloc){
  pi1=matrix(NA,nloc,nloc)
  for (i in 1:nrow(combo)){
    pi1[combo$x[i],combo$y[i]]=combo$prob[i]  
  }
  pi1
}

Comparison

The figure below illustrates that the random search is a terrible idea. In 0.3% of the simulations, the submarine was not found even after 10,000 steps! The idea of searching based on the prior distribution is not bad but it still occasionally fails to find the submarine whereas the Bayes search always found the submarine.

dat=data.frame(rand=res.rand,bayes=res.bayes,prior=res.prior)

#proportion of simulations that successfully found the submarine after 10,000 steps
apply(!is.na(dat),2,mean)

## 0.997 1.000 0.997

Importantly, the Bayes search generally resulted in shorter search times when compared to the prior distribution search.
