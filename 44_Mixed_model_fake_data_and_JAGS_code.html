As usual, we start by generating some fake data to see if our code is working. Here are the fake data:

rm(list=ls(all=T))
set.seed(1)

#import data
setwd('U:/uf/courses/bayesian course/2016/chapter11')
radon=read.csv('radon data gelman_hill.csv',as.is=T)

#set the "true" parameters
b1=-1
b2=-2
sig2=1
tau2=0.1
gamma=2

#generate the random intercepts
c.id=radon$county
K=length(unique(c.id))
b0=rnorm(K,mean=gamma,sd=sqrt(tau2))

#here are the covariates
x1=radon$floor
x2=radon$log.u

#generate log radon concentration
media=b0[c.id]+b1*x1+b2*x2
N=nrow(radon)
y=rnorm(N,mean=media,sd=sqrt(sig2))
radon$log.radon=y

#output the results
write.csv(radon,'fake data.csv',row.names=F)

Next, we write up our JAGS model. Here it is:

#JAGS code in "radon random.R"
model{
    for (i in 1:N){
        media[i] <- b0[c.id[i]]+b1*x1[i]+b2*x2[i]
        y[i]~dnorm(media[i],prec.sig2)
    }

    for (k in 1:K){
        b0[k] ~ dnorm(gamma,prec.tau2)
    }
    
    b1 ~ dnorm(0,1/100)
    b2 ~ dnorm(0,1/100)
    gamma ~ dnorm(0,1/100)
    prec.sig2 ~ dgamma(0.1,0.1)
    prec.tau2 ~ dgamma(0.1,0.1)
}

Finally, we invoke JAGS using the following R code:

rm(list=ls(all=T))
library(jagsUI)

setwd('U:/uf/courses/bayesian course/2016/chapter11')
radon=read.csv('fake data.csv',as.is=T)
#radon=read.csv('radon data gelman_hill.csv',as.is=T)

# MCMC settings 
ni <- 10000 #number of iterations
nt <- 5    #interval to thin 
nb <- 5000  #number of iterations to discard as burn-in
nc <- 3     #number of chains

#set parameters to monitor
params=c("b0","b1",'b2','gamma','prec.sig2','prec.tau2')

dat=list('y'=radon$log.radon,'x1'=radon$floor,'x2'=radon$log.u,'c.id'=radon$county,
         'N'=nrow(radon),'K'=length(unique(radon$county)))

#sets up model object
model2=jags(model.file="radon random.R",
            parameters.to.save=params,data=dat,n.chains=nc,
            n.burnin=nb,n.iter=ni,n.thin=nt,DIC=TRUE)

A comparison between the true and estimated parameters suggests that this model is working as expected. Top panels show traceplots, where each color corresponds to a different chain and horizontal red line is the true parameter value. Bottom panels show the marginal posterior distributions, where vertical blue continuous and dashed lines are mean and 95% credible intervals, respectively, while vertical red line is the true parameter value:

#inference on parameters
inference=function(gibbs,name1,true.value,nc){
  nsim=length(gibbs)/nc
  k1=matrix(gibbs,nsim,nc)
  plot(NA,NA,xlim=c(0,nsim),ylim=range(k1),main=name1,xlab='iterations',ylab='')
  for (i in 1:3) lines(1:nsim,k1[,i],col=i)
  abline(h=true.value,col='red',lwd=3)
  
  k2=density(k1)
  plot(k2,type='l',xlab='',main=name1,ylab='')
  z=quantile(k1,c(0.025,0.975))
  abline(v=z,col='blue',lty=3)
  abline(v=mean(k1),col='blue',lty=1)
  abline(v=true.value,col='red',lty=3)
}

inference(gibbs=model2$sims.list$b1,name1='b1',true.value=-1,nc)

inference(gibbs=model2$sims.list$b2,name1='b2',true.value=-2,nc)

inference(gibbs=model2$sims.list$gamma,name1='gamma',true.value=2,nc)
